class GridDraw {

    int BORDER_WIDTH (== width of border of around the grid --> so the panel)

    int VISIBLE_GRID_ROWS; (--> rows visible on the grid)
    int HIDDEN_GRID_ROWS; (--> rows hidden on the grid for shapespawning)

    int GRID_COLUMNS;
    int GRID_ROWS

    int BLOCK_SIZE (--> the size of the blocks on the grid)
    int SHADE_PIXELS (--> amount of pixels that on the block that makes a shadow

    int CENTERX_GRID (--> the middle of the grid on X)
    int CENTERY_GRID (--> the middle of the grid on Y)
    int PANEL_WIDTH (--> amount of colums * the size of blocks + BORDERWIDTH * 2)
    int PANEL_HEIGHT (--> amount of visible rows * size of blocks + BORDERWIDTH * 2)

    Game game;
    Shape[][] blocks;

    public GridDraw(Game game) {
        initialize game & blocks(--> amount of rows & colums)

        initialize GridDraw panel in background color black
    }

    void clear() {
        Loop through every block index and set its value to null
    }

    //All the checks for !shapeplacement!

    }

    //Add piece on grid

        loop through every tile of the piece (shapeDimension)
            if(shape.isBlock() (--> called from shape to check if that block is set to true in the shape))

    }

    //Checks if and how many lines there are cleared, another method is called for clearing the cleared lines
    int checkLines {
        int completedLines;

        //Loop through every row
            if(row is full and needs to be removed --> checkLine(row)) {
                completedLines++;
            }

        return completedLines
    }

    boolean checkLine(int line) {

        //Check through every column in the given row (fori)
            if(notOccupied(that column, given line) {
                return false --> because the row not fully occupied
            }

        //Loop that checks the row one above the give line and will loop till the rows equal to zero.
            //Loop that checks every colums per row
                setTile(column, nextRow(line +1) , and change this line with block on the current line

        //Line is cleared
        return true
    }

    //Check if a block is occupied
    boolean isOccupied(...) {
        returns true if the block on that coordinate is not null
    }

    //Set a block (coordinate) with a give block
    void setBlock(x, y, block) {
        shapes[][] = block
    }

    //Get a block (coordinate) by its column and row
    Shape getBlock(x,y) {
        return blocks[][];
    }

    //Draw a block on grid with give coordinates and shape
    void drawBlockNoColor(...){
        drawBlockWithColor(gets all the sorts of color from Shape, and also gives back coordinates);
    }

    void drawBlockWithColor(all sorts of color from shape and also the coordinates) {

        //Fills the entire block
        setFill(basecolor)
        drawRectangle(x,y,blocksize,blocksize)

        //Fills the south and west of block
        setFill(darkColor)
        drawRectangle(x, y+blocksize-shadewidth, blocksize, shadewidth)
        drawRectangle(x + blocksize - shadewidth, y, shadewidth, blocksize)

        //Fills the north and ease of block
        setFill(lightColor)
        //forloop is possible if we want diagnonal and then drawing lines with x or y everytime littlebit smaller, for now the same as above
        drawRectangle(x, y+shadewidth, blocksize, shadewidth)
        drawRectangle(x + shadewidth, y, shadewidth, blocksize)

    }

    void paint() {

        if (paused) {
            drawText("paused") at centerx - stringwidth and centery
        } else if (new game) {
            drawText("TETRIS")
            drawText("Press enter to play")
        } else if (gameover) {
            drawText("GAME OVER")
            drawText("Press enter to play again")
        } else {

            //Draw background on board
            for(x = 0, x < GRID_COLUMS; x++) {
                for(y = HIDDEN_GRID_ROWS; y < GRID_ROWS; y++) {
                    Shape block = getBlock(x,y);
                    if (block doesnt is NULL {
                        drawBlock(block, x * blocksize, (y - HIDDEN_GRID_ROWS) * blocksize)
                    }
                }
            }

            Shape shape = game.getShape()
            shapeCol = game.getShapeCol()
            shapeRow = game.getShapeRow()
            shapeRotation = game.getShapeRotation()

            //Draw piece
            fori(col = 0;  col < shape.dimension; col++){
                fori(row =0; row < shape.dimension; row++){
                    if(shapeRow + row >= 2 (--> When the row of the shape + the row of the loop is atleast 2, means it not out of the grid) && shape.isBlock(col,row,shapeRotation)) {
                        drawBlock(shape, (shapeCol + col) * blocksize, (shapeRow + row - HIDDEN_GRID_ROWS) * blocksize);
                    }
                }
            }

            //LOOPS FOR GHOSTPIECE --> First make working game...

            //Draw grid with lines (only the background color or blockcolor) --> less lagg
            setBorderColor(color)
            fori(x=0; x<GRID_COLUMNS; x++) {
                fori(y=0; y<VISIBLE_GRID_ROWS; y++) {
                    drawLine(0, y * blocksize, GRID_COLUMNS * blocksize, y * blocksize)
                    drawLine(x * blocksize, 0, x * blocksizem VISIBLE_GRID_ROWS * blocksize)
                }
            }
        }


        //Draw outline of panel
        drawLines * 4 (Left, Top, Right, Bottom)
    }

}
