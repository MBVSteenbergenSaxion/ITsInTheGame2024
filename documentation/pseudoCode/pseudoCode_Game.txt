    void keyboardEvent(KeyboardEvent keyboardEvent) {
        if (keyboardEvent.isKeyPressed()) {
            switch (keyboardEvent.getKeyCode()) {

                !//Wanneer de dropkey wordt gebruikt en de game is niet gepauzeerd en de cooldown voor het vallen van de volgende shape gelijk is aan 0,
                  dan word de cycle versneld naar 25 cycles per seconde ipv de cycle die wordt vastgezet in de clock methode.
                case (dropkey):
                    if (notPaused & dropCooldown == 0) {
                        clock.setCyclesPerSecond(25.0);
                        dropKeyPressed = true;
                    }
                    break;

                //Wanneer key naar links dan word de as één tile naar links verplaatst, aangegeven met col
                case (leftkey)
                    if (notPaused & gridDraw.isPossibleAndEmpty(...) {
                        currentCol--;
                    }
                    break;

                //Wanneer key naar rechts dan word de as één tile naar rechts verplaatst aangegeven met col
                case (rightkey)
                    if (notPaused & gridDraw.isPossibleAndEmpty(...) {
                        currentCol++;
                    }
                    break;

                //Wanneer key om tegen de klok in te roteren geklikt, dan roteert de shape op basis van de meegegeven rotatie
                  en wordt er gekeken of de huidige rotatie kleiner wordt dan 0, deze wordt dan terug geschreven naar 3
                case (rotateTegenKlokInKey)
                    if (notPaused & notRotationAntiClockWiseKeyPressed) {
                        int newRotation;
                        if (currentRotation == 0) {
                            newRotation = 3;
                        } else {
                            newRotation = currentRotation - 1;
                        }
                        rotateShape(newRotation);
                        rotationAntiClockWiseKeyPressed = true;
                    }
                    break;

                //Wanneer key om met de klok mee te roteren geklikt, dan roteert de shape op basis van de meegegeven rotatie
                  en wordt er gekeken of de huidige rotatie groter wordt dan 3, deze wordt dan terug geschreven naar 0
                case (rotateMetKlokMetKey)
                    if (notPaused & notRotationClockWiseKeyPressed) {
                        int newRotation;
                        if (currentRotation == 3) {
                            newRotation = 0;
                        } else {
                            newRotation = currentRotation + 1;
                        }
                        rotateShape(newRotation);
                        rotationClockWiseKeyPressed = true;
                    }

                //Wanneer key om te pauzeren wordt gedrukt, dan wordt er gekeken of de speler niet al gameover is en of er niet eerst een
                  nieuwe game moet worden gestart. Hierna krijgt de pauzeer boolean variable de tegengestelde waarde en wordt een methode vanuit
                  de clock klasse opgeroepen om het verder af te handelen
                case (pauseKey):
                    if(notGameOver & notNewGame) {
                        isPaused = !isPaused;
                        clock.setPaused(isPaused);
                    }
                    break;

                //Wanneer key startgame gedrukt, moet het gameover zijn of een nieuwe game zijn, hierna wordt een aparte methode aangeroepen om
                  het verder af te handelen
                case (startgameKey):
                    if (gameOver | newgame) {
                        resetGame();
                    }
                    break;
            }
        } else {
            switch (keyboardEvent.getKeyCode()) {
                case dropKey:
                    dropKeyPressed = false;
                case rotationClockWiseKey:
                    rotationClockWiseKeyPressed = false;
                case rotationAntiClockWiseKey:
                    rotationAntiClockWiseKeyPressed = false;
            }
        }
    }

    //Start de game running. Initialiseert alles en start gameloop
    void startGame() {

        //Initialiseert newGame boolean en gamespeed
        isNewGame = true;
        gameSpeed = ...;

        //Setup timer zonder dat ie begint te runnen voor er enter wordt gedruk
        clock = new Clock(gameSpeed);
        clock.setPaused(true);

        while(true) {
            //Krijg de tijd dat frame wordt gestart
            long start = System.nanoTime();

            //Update clock
            clock.update();

            //Wanneer cycle klaar is op de clock, dan kan de game worden geupdate en de shape omlaag worden geduwd
            if(clock.cycleDone()) {
                updateGame();
            }

            //Als het nodig is de dropcooldown af laten tellen
            if(dropcooldown > 0) {
                dropcooldown--;
            }

            //Spel tonen aan gebruiker
            updateRenderGame();

            //Als loop sneller is dan frame tijd, dan wordt er een sleep gedaan --> KAN NIET MET SAXION, moet aparte thread worden gedaan
              SAMEN kijken naar hoe we thread gaan implementeren ipv deze while loop in game.java!!!
            long delta = (System.nanoTime() - start) / 1000000
            if(delta < FRAME_TIME) {
                Thread.sleep(FRAME_TIME - delta);
            }
        }
    }

    //Laat shape roteren met de nieuwe rotatie waarde
    void rotateShape(int newRotation) {
        //Als de shape niet te roteren valt zonder buiten het grid te komen of samen met een andere shape in de background te vallen
          is het nodig om naast de shape te draaien ook een nieuwe as te maken. Dit wordt pas toegepast als dat nodig is
        newCol = currentCol;
        newRow = currentRow;

        //Krijg de omranding van de shape met de nieuwe terug op de grid
        int newLeftSide = currentShape.getLeftSide(newRotation);
        int newRightSide = currentShape.getRightSide(newRotation);
        int newTopSide = currentShape.getTopSide(newRotation);
        int newBottomSide = currentShape.getBottomSide(newRotation);

        //CHECK (nieuwe methode maken?) Wanneer de shape is te ver naar links of naar rechts,
        dat die buiten het grid valt moeten de coordinaten worden aangepast
        if (currentCol < -newLeftSide) {
            newCol = newLeftSide;
        } else if (currentCol + currentShape.getDimension() - newRightSide >= GridDraw.amountOfCol) {
            int adjustment = (currentCol + currentShape.getDimension() - newRightSide) - GridDraw.amountOfCol + 1;
            newCol = newCol - adjustment;
        }

        //CHECK (^) wanneer de shape te ver omhoog gaat of te laag naar beneden.
        if (currentRow < -newTopSide) {
            newRow = newTopSide;
        } else if (currentRow + currentShape.getDimension() - newBottomSide) >= GridDraw.amountOfRow) {
            int adjustment = (currentRow + currentShape.getDimension() - newRightSide) - GridDraw.amountOfRow + 1;
            newRow = newRow - adjustment;
        }

        //Als de nieuwe positie goed is dan de nieuwe variablen als huidige zetten (meegeven met de nieuwe variable dus)
        if (gridDraw.isPossibleAndEmpty(...) {
            currentRotation = newRotation;
            currentCol = newCol;
            currentRow = newRow;
        }
    }
